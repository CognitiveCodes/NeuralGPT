function neuralgpt_browse_search( WP_REST_Request $request ) {
    $searchQuery = $request->get_param( 'search_query' );
    $args = array(
        'post_type' => 'post',
        's' => $searchQuery
    );
    $query = new WP_Query( $args );
    $results = array();
    if ( $query->have_posts() ) {
        while ( $query->have_posts() ) {
            $query->the_post();
            $result = array(
                'title' => get_the_title(),
                'excerpt' => get_the_excerpt(),
                'link' => get_permalink()
            );
            array_push( $results, $result );
        }
    }
    wp_reset_postdata();
    return $results;
}
